#!/bin/bash

#1. If in ignore funtion:

InIgnore()
{
    extension=$1
    for i in ${!ARRAY[@]};
    do
        if [[ "${ARRAY[$i]}" = "${extension}" ]];
        then
            return 0;
        fi
    done

    return 1;
}

#2. recursively traverse all directory

traverse()
{
    cd "$1"

    for f in *
    do
        if [ -d "$f" ]; then
            traverse "$f"
        elif [ -f "$f" ]; then
            if [[ "$f" == *"."* ]]; then
                new_folder=`echo "$f" | cut -d '.' -f 2-`
            else
                new_folder="others"
            fi
            

            if InIgnore $new_folder; then
                ignore_count=$((ignore_count+1))
            else
                echo "$new_folder not in ignore"

                new_directory="$output_directory/$new_folder/"
                current_directory=`pwd`
                current_directory="$current_directory/$f"
                #echo "current directory: $current_directory"
                #echo "new directory: $new_directory"
                file_directory="$new_directory/desc_$new_folder.txt"

                if [ -d $new_directory ]; then
                    cp $current_directory $new_directory
                    echo $current_directory >> $file_directory 
                    echo "if er moddhe"
                else
                    mkdir $new_directory
                    cp $current_directory $new_directory
                    touch "$new_directory/desc_$new_folder.txt"
                    echo $current_directory > $file_directory
                    echo "else er moddhe"
                fi
            fi    
        fi
    
    done
}

#3. output csv file function:

outputCsv()
{

    touch "output.csv"
    #chmod u+x "output.csv"
    #install -b -m 755 /dev/null newfile
    #chmod 666 output.csv

    #chmod 666 outCsv>>!#:2

    echo "noice" > output.csv
    outString=""
    
    cd "$2"

    for f in `ls`
    do
        echo "$f"
        if [ -d $f ]; then
            cd $f
            cnt=-1
            for ff in `ls`
            do
                echo "$ff"
                cnt=$((cnt+1))
            done
            echo "$f, $cnt" >> outCsv
            outString="$outString
            $f, $cnt"
            cd ..
        fi
    done

    echo "ignored, $3" >> outCsv
    outString="$outString 
    ignored, $3"
    echo $outString
    echo "the file: "
    cat outCsv | echo 
    

}
