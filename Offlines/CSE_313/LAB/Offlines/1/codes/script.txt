#!/bin/bash

#1. If in ignore funtion:

InIgnore()
{
    extension=$1
    for i in ${!ARRAY[@]};
    do
        if (($extension == ${ARRAY[$i]}));
        then
            return 0;
        fi
    done

    return 1;
}

#2. recursively traverse all directory

traverse()
{
    cd "$1"

    for f in *
    do
        if [ -d "$f" ]; then
            traverse "$f"
        elif [ -f "$f" ]; then
            new_folder=`echo "$f" | cut -d '.' -f 2-`

            if InIgnore $new_folder; then
                ignore_count=`expr $ignore_count+1`
            else

                #if (($new_folder == "")); then
                 #   new_folder="others"
                #fi

                new_directory="$output_directory/$new_folder/"
                current_directory="$directory/$f"
                echo "current directory: $current_directory"
                file_directory="$new_directory/desc_$new_folder.txt"
            
                if [ -d $new_directory ]; then
                    cp $current_directory $new_directory
                    echo $current_directory >> $file_directory 
                else
                    mkdir $new_directory
                    cp $current_directory $new_directory
                    touch "$new_directory/desc_$new_folder.txt"
                    echo $current_directory > $file_directory
                fi
            fi    
        fi
    
    done
}

#main code 

filename="nothing"
directory="nothing"
invalid_filename=1
ignore_count=0

if (($# == 0)); then
    directory=`pwd`
    invalid_filename=0
    echo "File name not found, Enter filename"
elif (($# == 1)); then
    if [ -f $1 ]; then
        filename=$1
        directory=`pwd`
    elif [ -d $1]; then
        directory=$1
        invalid_filename=0
    else
        directory=`pwd`
        echo "File name not found, Enter working directory and filename as input"
        invalid_filename=0
    fi
elif (($# == 2)); then
    if [ -d $1 ]; then
        directory=$1
    else 
        echo "Not a directory"
        directory=`pwd`
    fi
    if [ -f $2 ]; then
        filename=$2
    else 
        echo "Not a valid filename"
        invalid_filename=0
    fi
else 
    echo "Invalid input, Enter working directory and valid filename as input"
    directory=`pwd`
fi



while (($invalid_filename == 0))
do
    echo "Enter valid file name: "
    read $filename

    if [ -f filename ]; then
        invalid_filename=1
    fi
done

echo $filename
echo $directory

#read ignore file extensions

cnt=0
while read line; do
    cnt=`expr $cnt+1` 
    ARRAY[$cnt]=$line
done < $filename

echo ${ARRAY[*]}
echo "now you shall perish"

#create output directory:

cd $directory
cd ..
output_directory=`pwd`
output_directory="$output_directory/output_directory/"
mkdir $output_directory

#call traverse function:
cd $directory
traverse $directory
